## 🎯 功能 1：權限系統（資料庫層）

### 目標
在 `members` 表新增 `role` 欄位，支援 4 種權限：
- `開發者`：完整權限
- `管理者`：管理權限
- `負責人`：部分管理權限
- `一般人`：基本權限（預設）

### 實作步驟

#### 步驟 1.1：執行資料庫遷移
```sql
-- 檔案：database/add-role-column.sql（已存在）
-- 在 Supabase SQL Editor 中執行

ALTER TABLE members ADD COLUMN IF NOT EXISTS role VARCHAR(20) DEFAULT '一般人';
COMMENT ON COLUMN members.role IS '權限角色：開發者、管理者、負責人、一般人';
CREATE INDEX IF NOT EXISTS idx_members_role ON members(role);

-- 新增檢查約束，確保只能填入有效的角色
DO $$
BEGIN
  IF NOT EXISTS ( SELECT 1 FROM pg_constraint WHERE conname = 'members_role_check' ) THEN
    ALTER TABLE members ADD CONSTRAINT members_role_check CHECK (role IN ('開發者', '管理者', '負責人', '一般人'));
  END IF;
END $$;
```

#### 步驟 1.2：設定你的開發者權限

```sql
-- 將你的帳號設為開發者

-- 驗證
```

## 🎯 功能 2：生日行程自動同步

### 目標

### 實作步驟

```javascript
// 檔案：server/services/calendarService.js

// 新增函式：同步單一用戶的生日行程
const syncMemberBirthdayEvents = async (member) => {
  const name = (member.name || '').trim();
  const parsed = parseBirthdayMonthDay(member.birthday);
  
  if (!name || !parsed) {
    console.log('⚠️ 成員無有效姓名或生日，跳過同步');
    return 0;
  }

  const { month, day } = parsed;
  const correctTitle = `🎂 ${name}生日`;
  const currentYear = new Date().getFullYear();
  const yearsToCheck = [currentYear, currentYear + 1];

  let totalSynced = 0;
  const eventDbService = require('./eventDbService');

  // 步驟 1: 掃描並清理重複的生日行程
  for (const year of yearsToCheck) {
    const birthdayDate = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const timeMin = `${birthdayDate}T00:00:00+08:00`;
    const timeMax = `${birthdayDate}T23:59:59+08:00`;
    
    try {
      // 從 Google Calendar 取得該日期的所有行程
      const events = await calendar.events.list({
        calendarId: CALENDAR_ID,
        timeMin,
        timeMax,
        singleEvents: true,
      });

      const birthdayEvents = (events.data.items || []).filter(e => 
        e.summary && e.summary.includes(name) && e.summary.includes('生日')
      );

      const canonicalId = `birthday_${member.lineId}_${year}`;
      const canonical = birthdayEvents.find(e => e.id === canonicalId);
      const duplicates = birthdayEvents.filter(e => e.id !== canonicalId);

      // 刪除重複的行程
      for (const dup of duplicates) {
        try {
          await calendar.events.delete({
            calendarId: CALENDAR_ID,
            eventId: dup.id,
          });
          console.log(`✅ 已刪除重複的生日行程: ${dup.id}`);
        } catch (err) {
          console.warn(`⚠️ 刪除重複行程失敗: ${err.message}`);
        }
      }
    } catch (error) {
      console.warn(`⚠️ 掃描 ${year} 年生日行程失敗:`, error.message);
    }
  }

  // 步驟 2: 同步正確的生日行程
  for (const year of yearsToCheck) {
    const birthdayDate = `${year}-${String(month).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const birthdayId = `birthday_${member.lineId}_${year}`;

    try {
      // 檢查行程是否存在
      let existingEvent;
      try {
        existingEvent = await calendar.events.get({
          calendarId: CALENDAR_ID,
          eventId: birthdayId,
        });
      } catch (error) {
        if (error.code !== 404) throw error;
      }

      if (existingEvent) {
        // 檢查標題和日期是否正確
        const needsUpdate = 
          existingEvent.data.summary !== correctTitle ||
          !existingEvent.data.start.date ||
          existingEvent.data.start.date !== birthdayDate;

        if (needsUpdate) {
          // 更新行程
          const updated = await calendar.events.update({
            calendarId: CALENDAR_ID,
            eventId: birthdayId,
            requestBody: {
              summary: correctTitle,
              start: { date: birthdayDate },
              end: { date: birthdayDate },
              description: `${name} 的生日`,
            },
          });

          await eventDbService.upsertEvents([{
            id: birthdayId,
            title: correctTitle,
            start: birthdayDate,
            end: birthdayDate,
            allDay: true,
            type: '生日',
            isBirthdayEvent: true,
          }]);

          totalSynced++;
          console.log(`✅ 已更新 ${year} 年生日行程: ${correctTitle}`);
        }
      } else {
        // 建立新行程
        const created = await calendar.events.insert({
          calendarId: CALENDAR_ID,
          requestBody: {
            id: birthdayId,
            summary: correctTitle,
            start: { date: birthdayDate },
            end: { date: birthdayDate },
            description: `${name} 的生日`,
          },
        });

        await eventDbService.upsertEvents([{
          id: birthdayId,
          title: correctTitle,
          start: birthdayDate,
          end: birthdayDate,
          allDay: true,
          type: '生日',
          isBirthdayEvent: true,
        }]);

        totalSynced++;
        console.log(`✅ 已建立 ${year} 年生日行程: ${correctTitle}`);
      }
    } catch (error) {
      console.error(`❌ 同步 ${year} 年生日行程失敗:`, error.message);
    }
  }

  return totalSynced;
};
```

## 🎯 功能 3：並行讀取策略

### 目標
優化行事曆 API 效能：
1. 立即從 Supabase 資料庫讀取並回應
2. 背景非同步同步 Google Calendar 資料

### 實作步驟

#### 步驟 3.1：更新 calendar.js 路由

```javascript
// 檔案：server/routes/calendar.js

// 新增背景同步函式（放在檔案最後）
async function syncCalendarInBackground(timeMin, timeMax) {
  try {
    console.log(`🔄 背景同步開始: ${timeMin} ~ ${timeMax}`);
    
    // 從 Google Calendar 取得行程
    const response = await calendarService.listGroupEvents(timeMin, timeMax);
    const events = response.data.items || [];
    
    if (events.length === 0) {
      console.log('ℹ️ Google Calendar 沒有新行程');
      return;
    }

    // 轉換並存入資料庫
    const eventDbService = require('../services/eventDbService');
    const eventsToUpsert = events.map(e => ({
      id: e.id,
      title: e.summary || '',
      description: e.description || '',
      start: e.start.dateTime || e.start.date,
      end: e.end.dateTime || e.end.date,
      allDay: !e.start.dateTime,
      location: e.location || '',
      type: e.extendedProperties?.private?.type || '活動',
      isBirthdayEvent: e.id?.startsWith('birthday_') || false,
    }));

    await eventDbService.upsertEvents(eventsToUpsert);
    
    // 刪除在 Google Calendar 中已被刪除的行程
    const gcalIds = new Set(events.map(e => e.id));
    const dbEvents = await eventDbService.getEventsByRange(timeMin, timeMax);
    const toDelete = dbEvents
      .filter(e => !e.isBirthdayEvent && !gcalIds.has(e.id))
      .map(e => e.id);
    
    if (toDelete.length > 0) {
      await eventDbService.deleteEvents(toDelete);
      console.log(`🗑️ 已刪除 ${toDelete.length} 個過期行程`);
    }

    console.log(`✅ 背景同步完成: ${eventsToUpsert.length} 個行程`);
  } catch (error) {
    console.error('❌ 背景同步失敗:', error.message);
  }
}

// 修改 GET /api/calendar/month 路由
router.get('/month', optionalLineUser, async (req, res) => {
  try {
    const year = parseInt(req.query.year);
    const month = parseInt(req.query.month);
    const type = req.query.type || null;

    if (!year || !month || month < 1 || month > 12) {
      return res.status(400).json({
        success: false,
        message: '請提供有效的年份和月份',
      });
    }

    // 1. 立即從資料庫讀取（快速回應）
    const events = await eventDbService.getEventsByMonth(year, month, type);

    // 2. 計算時間範圍
    const timeMin = `${year}-${String(month).padStart(2, '0')}-01T00:00:00+08:00`;
    const lastDay = new Date(year, month, 0).getDate();
    const timeMax = `${year}-${String(month).padStart(2, '0')}-${String(lastDay).padStart(2, '0')}T23:59:59+08:00`;

    // 3. 背景同步（不阻塞回應）
    syncCalendarInBackground(timeMin, timeMax).catch(err => {
      console.warn('⚠️ 背景同步略過（不影響前端）:', err.message);
    });

    // 4. 立即回應
    res.json({
      success: true,
      data: events,
    });
  } catch (error) {
    console.error('取得月份行程錯誤:', error);
    res.status(500).json({
      success: false,
      message: error.message || '取得月份行程失敗',
    });
  }
});

// 修改 GET /api/calendar/today 路由（同樣邏輯）
router.get('/today', optionalLineUser, async (req, res) => {
  try {
```

```javascript
/**
 * POST /api/admin/sync-all-birthdays
 * 強制同步所有用戶的生日行程（開發者專用）
 */
router.post('/sync-all-birthdays', verifyLineUser, verifyAdmin, async (req, res) => {
  try {
    const members = await memberDbService.getAllMembers();
    const membersWithBirthday = members.filter(m => m.birthday);

    let totalSynced = 0;
    const results = [];

    for (const member of membersWithBirthday) {
      try {
        const count = await calendarService.syncMemberBirthdayEvents(member);
        totalSynced += count;
        results.push({
          name: member.name,
          synced: count,
          success: true,
        });
      } catch (error) {
        results.push({
          name: member.name,
          synced: 0,
          success: false,
          error: error.message,
        });
      }
    }

    res.json({
      success: true,
      data: {
        totalMembers: membersWithBirthday.length,
        totalSynced,
        results,
      },
      message: `已同步 ${totalSynced} 個生日行程`,
    });
  } catch (error) {
    console.error('同步所有生日錯誤:', error);
    res.status(500).json({
      success: false,
      message: error.message || '同步所有生日失敗',
    });
  }
});

/**
 * GET /api/admin/members
 * 取得所有成員列表（含角色）（開發者專用）
 */
router.get('/members', verifyLineUser, verifyAdmin, async (req, res) => {
  try {
    const members = await memberDbService.getAllMembers();
    
    res.json({
      success: true,
      data: members,
    });
  } catch (error) {
    console.error('取得成員列表錯誤:', error);
    res.status(500).json({
      success: false,
      message: error.message || '取得成員列表失敗',
    });
  }
});

/**
 * PUT /api/admin/members/:lineId/role
 * 更新成員角色（開發者專用）
 */
router.put('/members/:lineId/role', verifyLineUser, verifyAdmin, async (req, res) => {
  try {
    const { lineId } = req.params;
    const { role } = req.body;

    if (!role) {
      return res.status(400).json({
        success: false,
        message: '請提供角色',
      });
    }

    const member = await memberDbService.updateMemberRole(lineId, role);

    res.json({
      success: true,
      data: member,
      message: '角色更新成功',
    });
  } catch (error) {
    console.error('更新角色錯誤:', error);
    res.status(500).json({
      success: false,
      message: error.message || '更新角色失敗',
    });
  }
});

module.exports = router;
```
